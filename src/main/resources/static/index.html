SQS Polling for new messages


@MessageMapping(QUEUE_NAME)
private void receiveMessage(MessageToProcess message,
                            @Header("ApproximateFirstReceiveTimestamp") String approximateFirstReceiveTimestamp) {
    try {
        sendMessageToClient(new MessageWithReceiveTimestamp(message, approximateFirstReceiveTimestamp));
    } catch (IOException e) {
        LOG.error("Was not able to push the message to the client.", e);
    }
}
